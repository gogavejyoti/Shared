function toggleRowGroup(groupIndex) {

    let sheet = F1.luckysheetfile[F1.currentSheetIndex]

    if (!sheet.rowGroups) return

    let group = sheet.rowGroups[groupIndex]

    if (!group) return

    // toggle state
    group.collapsed = !group.collapsed

    for (let r = group.start + 1; r <= group.end; r++) {

        if (group.collapsed)
            hideRow(r)
        else
            showRow(r)

    }

    // refresh grid
    luckysheet.refresh()

}
