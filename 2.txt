    function loadWorkbook() {

        $.ajax({
            url: '/ResourcePlanner/GetWorkbook?id=' + planId + '&vertical=' + verticalName,
            method: 'GET',
            success: function (data) {
                $('#loader-overlay').css('display', 'none');
                if (data && data.sheets && data.sheets.length > 0) {
                    decompressData(data.sheets);
                    initDocumentLock($("#planId").val(), $('#hdnUserName').val(), $('#isUserWriteAccess').val() === "True");
                    currentSheets = data.sheets.map(s => ({
                        name: s.name,
                        type: s.type,
                        rpPlanSheetId: s.rpPlanSheetId
                    }));
                    // Set workbook and launch Luckysheet
                    luckysheet.create({
                        container: 'luckysheet',
                        exportFileName: $('.lob-name').text(),
                        lang: 'en',
                        locale: 'en-US',
                        plugins: [{ name: 'chart' }],
                        data: data.sheets.map(s => {
                            let parsed = JSON.parse(s.jsonData);
                            const out = {
                                ...parsed,
                                name: s.name
                            };
                            parsed = null;
                            s.jsonData = undefined;
                            return out;
                        }),
                        onWorkbookLoaded: function () {
                            logMissingCalcChain();
                        },
                        hook: {
                            workbookCreateAfter: function () {
                            },
                            cellUpdated: function (r, c, oldValue, newValue) {
                                if (oldValue != newValue && !isSheetChanged)
                                    isSheetChanged = true;
                            },
                        }

                    });



                    $('#saveContainer').show();
                } else {
                    alert("No sheets found for this plan.");
                }
            },
            error: function (err) {
                $('#loader-overlay').css('display', 'none');
                console.error("Error loading plan", err);
                alert("Failed to load plan data.");
            }
        });

    }
