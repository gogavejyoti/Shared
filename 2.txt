public string GenerateUserSummary(List<ExecutiveWeekNarration> weekNarrations)
{
    var sb = new StringBuilder();

    foreach (var narration in weekNarrations)
    {
        sb.AppendLine($"Week: {narration.Week}");
        sb.AppendLine($"Summary: {narration.Header}");

        foreach (var point in narration.Points)
        {
            string lob = string.IsNullOrEmpty(point.LOB) ? "LOB" : point.LOB;
            string commentary = point.Commentary?.Trim();
            string valueStr = point.Value.HasValue ? $" (Value: {point.Value.Value:F1})" : "";

            if (!string.IsNullOrWhiteSpace(commentary))
            {
                sb.AppendLine($"- {lob}: {commentary}{valueStr}");
            }
        }

        sb.AppendLine();
    }

    return sb.ToString();
}
