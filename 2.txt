h.setCellValue = function(row, col, value, triggerUpdate = true) {
    const oldCell = $.extend(true, {}, this.flowdata[row][col]);
    let cell = this.flowdata[row][col] || {};

    delete cell.f;
    delete cell.spl;
    cell.v = value;
    if (!cell.ct) cell.ct = {};
    cell.ct.t = "g";
    cell.ct.fa = "General";

    this.flowdata[row][col] = cell;
    const newFlowData = $.extend(true, {}, this.flowdata);

    if (triggerUpdate) {
        Ye(newFlowData, [{ row: [row, row], column: [col, col] }], { data: newFlowData }, true);
        Je.createHookFunction("cellUpdated", row, col, oldCell, this.flowdata[row][col], true);
    }
};
