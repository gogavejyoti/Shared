var calcChain = luckysheet.getSheet(sheetIndex).calcChain;

calcChain.forEach(function(pos) {
    var r = pos.row;
    var c = pos.col;
    var cell = luckysheet.getSheet(sheetIndex).flowdata[r][c];
    if (cell && cell.f) {
        try {
            cell.v = luckysheet.formula.getcellvalue(r, c, sheetIndex);
        } catch(e){
            cell.v = null;
        }
    }
});

// Redraw only viewport
luckysheet.refresh({ type: "view" });
