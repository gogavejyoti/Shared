
        [HttpPost]
        public async Task<IActionResult> SaveSheets([FromBody] List<SaveSheetDto> sheets)
        {
            if (sheets == null || !sheets.Any())
                return BadRequest("Invalid sheet data.");
            var userId = User.GetIdByEmail();
            await _resourcePlannerService.SaveSheetsAsync(sheets, _vertical, userId);
            return Ok();
        }
