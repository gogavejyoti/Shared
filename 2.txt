using System;
using System.Data;
using System.Globalization;

public static class DataTableHelper
{
    public static void MultiplyPercentageColumnsBy100(DataTable table)
    {
        if (table == null) throw new ArgumentNullException(nameof(table));

        foreach (DataColumn col in table.Columns)
        {
            // Case-insensitive check for "percentage" in column name
            if (col.ColumnName.IndexOf("percentage", StringComparison.OrdinalIgnoreCase) >= 0)
            {
                for (int i = 0; i < table.Rows.Count; i++)
                {
                    var value = table.Rows[i][col];

                    if (value != DBNull.Value)
                    {
                        if (double.TryParse(value.ToString(), NumberStyles.Any, CultureInfo.InvariantCulture, out double num))
                        {
                            table.Rows[i][col] = num * 100;
                        }
                    }
                }
            }
        }
    }
}
