function renderGroupButton(rowHeader, groupIndex, group) {

    let btn = document.createElement("span")

    btn.className = "row-group-toggle"

    btn.innerText = group.collapsed ? "+" : "-"

    btn.style.marginLeft = "4px"
    btn.style.cursor = "pointer"

    btn.onclick = function (e) {

        e.stopPropagation()

        toggleRowGroup(groupIndex)

    }

    rowHeader.appendChild(btn)

}
