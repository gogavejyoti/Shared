// The main record representing a row from your SQL result
public class WorkforceData
{
    public string Week { get; set; }
    public string Geo { get; set; }
    public string Client { get; set; }
    public string Project { get; set; }
    public int SchHC { get; set; }
    public int ActSchHC { get; set; }
    public int Delta => SchHC - ActSchHC; // Calculated
    public string Program { get; set; }
    public string OUDescription { get; set; }
    public string ProjectID { get; set; }
    public string BillableStatus { get; set; } // "Yes" or "No"
}

// Object sent from Frontend to API
public class DashboardRequest
{
    public List<string> Weeks { get; set; }
    public List<string> Geos { get; set; }
    public List<string> Clients { get; set; }
    public List<string> Projects { get; set; }
    public List<string> Programs { get; set; }
    public List<string> OUs { get; set; }
    public List<string> Descriptions { get; set; }
    public List<string> ProjectIDs { get; set; }
}



[Route("api/Workforce")]
public class WorkforceController : Controller
{
    [HttpPost]
    [Route("GetData")]
    public JsonResult GetData([FromBody] DashboardRequest filters)
    {
        // 1. Fetch data from your database (Mocked for this example)
        var allData = GetSQLData(); 

        // 2. Apply dynamic filtering
        var filtered = allData.Where(x => 
            (filters.Weeks == null || !filters.Weeks.Any() || filters.Weeks.Contains(x.Week)) &&
            (filters.Geos == null || !filters.Geos.Any() || filters.Geos.Contains(x.Geo))
            // ... add other 6 filters here
        ).ToList();

        // 3. Shape the data for the 5 charts
        var response = new {
            TableData = filtered,
            KPIs = new {
                TotalSch = filtered.Sum(x => x.SchHC),
                TotalAct = filtered.Sum(x => x.ActSchHC),
                TotalDelta = filtered.Sum(x => x.Delta),
                BillableRate = filtered.Count > 0 ? (filtered.Count(x => x.BillableStatus == "Yes") * 100 / filtered.Count) : 0
            },
            TrendChart = filtered.GroupBy(x => x.Week).Select(g => new { 
                label = g.Key, 
                sch = g.Sum(s => s.SchHC), 
                act = g.Sum(a => a.ActSchHC) 
            }).OrderBy(x => x.label),
            GeoChart = filtered.GroupBy(x => x.Geo).Select(g => new { 
                label = g.Key, 
                value = g.Sum(s => s.ActSchHC) 
            })
        };

        return Json(response);
    }
}

        function refreshDashboard() {
            const filters = {
                Weeks: $('#f-week').val(),
                Geos: $('#f-geo').val(),
                Clients: $('#f-client').val(),
                Projects: $('#f-project').val(),
                Programs: $('#f-program').val(),
                OUs: $('#f-ou').val(),
                Descriptions: $('#f-desc').val(),
                ProjectIDs: $('#f-pid').val()
            };

            $.ajax({
                url: '/api/Workforce/GetData',
                type: 'POST',
                contentType: 'application/json',
                data: JSON.stringify(filters),
                success: function(res) {
                    // Update KPIs
                    $('#kpi-sch').text(res.KPIs.TotalSch);
                    $('#kpi-act').text(res.KPIs.TotalAct);
                    $('#kpi-delta').text(res.KPIs.TotalDelta);
                    $('#kpi-bill').text(res.KPIs.BillableRate + '%');

                    // Update Table
                    $('#mainTable').DataTable().clear().rows.add(res.TableData).draw();

                    // Update Charts
                    trendChart.updateSeries([{ name:'Sch', data: res.TrendChart.map(x=>x.sch) }, { name:'Act', data: res.TrendChart.map(x=>x.act) }]);
                    geoChart.updateSeries(res.GeoChart.map(x=>x.value));
                    geoChart.updateOptions({ labels: res.GeoChart.map(x=>x.label) });
                }
            });
        }

        function initCharts() {
            // Trend
            trendChart = new ApexCharts(document.querySelector("#chart-trend"), {
                series: [], chart: { type:'area', height:300, toolbar:{show:false}, theme:{mode:'dark'} },
                stroke: { curve:'smooth', width:2 }, colors:['#3699FF', '#0BB783']
            });
            trendChart.render();

            // Geo
            geoChart = new ApexCharts(document.querySelector("#chart-geo"), {
                series: [], chart: { type:'donut', height:300 }, labels: [], theme:{mode:'dark'}
            });
            geoChart.render();

            // Other 3 chart initializations go here... (using same logic)
        }

        function initTable() {
            $('#mainTable').DataTable({
                columns: [
                    { className: 'custom-drill', data: null, orderable: false, defaultContent: '<i class="fas fa-plus-circle text-primary"></i>' },
                    { data: 'Week' }, { data: 'Geo' }, { data: 'Client' }, { data: 'Project' },
                    { data: 'SchHC' }, { data: 'ActSchHC' }, { data: 'Delta' }
                ],
                dom: 'rtp', pageLength: 10
            });

            $('#mainTable tbody').on('click', 'td.custom-drill', function() {
                var tr = $(this).closest('tr');
                var row = $('#mainTable').DataTable().row(tr);
                if (row.child.isShown()) {
                    row.child.hide(); tr.removeClass('shown');
                } else {
                    const d = row.data();
                    row.child(`<div class="drilldown-container"><strong>Program:</strong> ${d.Program} | <strong>OU:</strong> ${d.OUDescription}</div>`).show();
                    tr.addClass('shown');
                }
            });
