getSelectedCellsData: function () {
    var selectedCells = this.host.find(".cellStyleSelected");
    var data = [];

    selectedCells.each((i, cellDiv) => {
        var cell = a(cellDiv).data("pivot-cell"); // Assuming you can access the cell info via data attr
        if (cell && cell.pivotRow && cell.pivotColumn) {
            var valueObj = this.source.getCellValue(cell.pivotRow.adapterItem, cell.pivotColumn.adapterItem);
            var rowText = cell.pivotRow.adapterItem.text;
            var colText = cell.pivotColumn.adapterItem.text;
            var value = valueObj ? valueObj.formattedValue : '';
            data.push([rowText, colText, value]);
        }
    });

    return data;
}
