files.forEach((sheet, sheetIdx) => {
    const chain = sheet.calcChain;
    if (!chain) return;

    chain.forEach(item => {
        const r = item.r;
        const c = item.c;

        const cell = sheet.data[r]?.[c];
        if (!cell || !cell.f) return;

        if (/TODAY\(|NOW\(/i.test(cell.f)) {
            luckysheet.updateCellFormula(sheet, r, c, cell.f, true);
        }
    });
});
