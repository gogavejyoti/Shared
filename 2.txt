function Ye(e, n, t, l = true, a = true, bulkupdate = false, oldData = undefined) {
    e = e || h.flowdata;
    n = n || h.luckysheet_select_save;
    n = JSON.parse(JSON.stringify(n));

    clearTimeout(Rr);
    t = t || {};

    let o = t.cfg,
        s = t.calc,
        u = t.RowlChange,
        d = t.cdformat,
        f = t.dataVerification,
        m = t.dynamicArray,
        g = t.hyperlink;

    let y = h.luckysheetfile[K(h.currentSheetIndex)];

    // 🔥 OPTIMIZED UNDO STORAGE
    if (h.clearjfundo && !bulkupdate) {
        h.jfundo.length = 0;

        // Only capture changed cells instead of full sheet
        const changedData = [];

        for (let i = 0; i < n.length; i++) {
            let r1 = n[i].row[0];
            let r2 = n[i].row[1];
            let c1 = n[i].column[0];
            let c2 = n[i].column[1];

            for (let r = r1; r <= r2; r++) {
                for (let c = c1; c <= c2; c++) {
                    changedData.push({
                        r,
                        c,
                        old: h.flowdata[r]?.[c],
                        new: e[r]?.[c]
                    });
                }
            }
        }

        h.jfredo.push({
            type: "datachange",
            sheetIndex: h.currentSheetIndex,

            // store only diff
            diff: changedData,

            config: o || h.config,
            curConfig: o || h.config,

            calc: s || y.calcChain,
            curCalc: s,

            cdformat: d || y.luckysheet_conditionformat_save,
            curCdformat: d,

            dataVerification: f || y.dataVerification,
            curDataVerification: f,

            dynamicArray: m || y.dynamicArray,
            curDynamicArray: m,

            hyperlink: g || y.hyperlink,
            curHyperlink: g,

            range: n
        });

        // Limit undo stack
        const UNDO_LIMIT = 3; // reduced further
        if (h.jfredo.length > UNDO_LIMIT) {
            h.jfredo.shift();
        }
    }

    // ✅ APPLY DATA (NO CLONE)
    h.flowdata = e;
    y.data = h.flowdata;

    // Config updates (only if changed)
    if (o) {
        h.config = o;
        y.config = o;
        ne.saveParam("all", h.currentSheetIndex, o, { k: "config" });

        if (u) {
            Ct(h.flowdata.length, h.flowdata[0].length);
        }
    }

    if (s) {
        y.calcChain = s;
        ne.saveParam("all", h.currentSheetIndex, s, { k: "calcChain" });
    }

    if (d && d.length) {
        y.luckysheet_conditionformat_save = d;
        ne.saveParam("all", h.currentSheetIndex, d, { k: "luckysheet_conditionformat_save" });
    }

    if (f && Object.keys(f).length) {
        Xe.dataVerification = f;
        y.dataVerification = f;
        ne.saveParam("all", h.currentSheetIndex, f, { k: "dataVerification" });
    }

    if (m) {
        y.dynamicArray = m;
        ne.saveParam("all", h.currentSheetIndex, m, { k: "dynamicArray" });
    }

    if (g) {
        y.hyperlink = g;
        Cl.hyperlink = g;
        ne.saveParam("all", h.currentSheetIndex, g, { k: "hyperlink" });
    }

    // 🔁 Recalculate only affected cells
    for (let i = 0; i < n.length; i++) {
        let r = n[i].row[0];
        let c = n[i].column[0];

        let cell = h.flowdata[r]?.[c];

        if (cell && cell.spl && cell.f) {
            try {
                window.luckysheetCurrentRow = r;
                window.luckysheetCurrentColumn = c;
                window.luckysheetCurrentFunction = cell.f;

                let parsed = $.trim(p.functionParserExe(cell.f));
                cell.spl = new Function("return " + parsed)();
            } catch (err) {
                console.error("Formula error:", err);
            }
        }

        if (ne.allowUpdate) {
            ne.historyParam(h.flowdata, h.currentSheetIndex, n[i]);
        }

        if (typeof h.chartparam.jfrefreshchartall === "function") {
            h.chartparam.jfrefreshchartall(
                h.flowdata,
                n[i].row[0],
                n[i].row[1],
                n[i].column[0],
                n[i].column[1]
            );
        }
    }

    if (l) {
        gc(n, h.currentSheetIndex, e, oldData);
    }

    if (a) {
        Rr = setTimeout(() => He(), 1);
    }

    tt();
    window.luckysheet_getcelldata_cache = null;
}
