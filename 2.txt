 private string GetSeverity()
    {
        if (Metric == "FTE Delta")
            return Math.Abs(Actual) > 40 ? "High" :
                   Math.Abs(Actual) > 20 ? "Medium" : "Low";

        double delta = Math.Abs(Actual - Target);
        if (delta >= 10) return "High";
        if (delta >= 5) return "Medium";
        return "Low";
    }
