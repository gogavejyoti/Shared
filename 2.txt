initialFreezen: function (e) {
    let n = this;
    n.frozenTofreezen();

    let t = h.luckysheetfile[K(e)];

    // -------- HORIZONTAL FREEZE (ROWS) --------
    if (
        t.freezen &&
        t.freezen.horizontal &&
        Array.isArray(t.freezen.horizontal.freezenhorizontaldata)
    ) {
        // ✅ Use index at position [1]
        const rowIndex = t.freezen.horizontal.freezenhorizontaldata[1];

        n.createFreezenHorizontal(
            rowIndex,
            t.freezen.horizontal.top
        );
    } else {
        n.cancelFreezenHorizontal(e);
    }

    // -------- VERTICAL FREEZE (COLUMNS) --------
    if (
        t.freezen &&
        t.freezen.vertical &&
        Array.isArray(t.freezen.vertical.freezenverticaldata)
    ) {
        // ✅ Use index at position [1]
        const colIndex = t.freezen.vertical.freezenverticaldata[1];

        n.createFreezenVertical(
            colIndex,
            t.freezen.vertical.left
        );
    } else {
        n.cancelFreezenVertical(e);
    }

    n.createAssistCanvas();
}
