    <script>
        // Global Instances
        let trendChart, geoChart, deltaChart, billChart, hrChart, mainTable;

        // Base Configuration for Charts (Sober Theme)
        const baseChartConfig = {
            fontFamily: 'Inter',
            theme: { mode: 'dark' },
            chart: { background: 'transparent', toolbar: { show: false } },
            grid: { borderColor: '#2B2B40' },
            dataLabels: { enabled: false },
            tooltip: { theme: 'dark' },
            xaxis: { labels: { style: { colors: '#92929f' } } },
            yaxis: { labels: { style: { colors: '#92929f' } } },
            legend: { labels: { colors: '#fff' } }
        };

        $(document).ready(function() {
            // 1. Initialize Components
            $('.select2').select2({ width: '100%', placeholder: 'All...', allowClear: true });
            
            initCharts();
            initTable();

            // 2. Fetch Initial Data
            refreshDashboard();

            // 3. Trigger Refresh on Filter Change
            $('.select2').on('change', function() {
                refreshDashboard();
            });
        });

        // --- AJAX LOGIC ---
        function refreshDashboard() {
            // Collect Filters
            const filters = {
                Weeks: $('#f-week').val(),
                Geos: $('#f-geo').val(),
                Clients: $('#f-client').val(),
                Projects: $('#f-project').val(),
                Programs: $('#f-program').val(),
                OUs: $('#f-ou').val(),
                Descriptions: $('#f-desc').val(),
                ProjectIDs: $('#f-pid').val()
            };

            // AJAX Call to your MVC API
            $.ajax({
                url: '/api/Workforce/GetData', // Your Controller Endpoint
                type: 'POST',
                contentType: 'application/json',
                data: JSON.stringify(filters),
                success: function(response) {
                    updateUI(response);
                },
                error: function(err) {
                    console.error("Error fetching data:", err);
                    // Optional: Show error toast
                }
            });
        }

        function updateUI(data) {
            // 1. Update KPIs (Handling Case Sensitivity)
            // Assumes API returns { KPIs: { TotalSch: 100... }, ... }
            const kpi = data.KPIs || data.kpis; 
            $('#kpi-sch').text(kpi.TotalSch || kpi.totalSch);
            $('#kpi-act').text(kpi.TotalAct || kpi.totalAct);
            $('#kpi-delta').text(kpi.TotalDelta || kpi.totalDelta);
            // Example of mapping "Leaves" if your API sends it
            $('#kpi-leave').text(kpi.TotalLeaves || kpi.totalLeaves || 0);

            // 2. Update Charts
            
            // Trend
            const trend = data.TrendChart || data.trendChart;
            trendChart.updateSeries([
                { name: 'Scheduled', data: trend.map(x => x.sch) },
                { name: 'Actual', data: trend.map(x => x.act) }
            ]);
            trendChart.updateOptions({ xaxis: { categories: trend.map(x => x.label) }});

            // Geo
            const geo = data.GeoChart || data.geoChart;
            geoChart.updateSeries(geo.map(x => x.value));
            geoChart.updateOptions({ labels: geo.map(x => x.label) });

            // Billable (Radial)
            const billRate = kpi.BillableRate || kpi.billableRate;
            billChart.updateSeries([billRate]);

            // Top Deltas (Bar)
            // Assuming API sends { DeltaChart: [{ label: 'Proj A', value: 10 }...] }
            if(data.DeltaChart || data.deltaChart) {
                const dChart = data.DeltaChart || data.deltaChart;
                deltaChart.updateSeries([{ data: dChart.map(x => x.value) }]);
                deltaChart.updateOptions({ xaxis: { categories: dChart.map(x => x.label) }});
            }

            // HR Stacked
            // Assuming API sends { HRChart: { Attrition: [1,2], Leaves: [3,4], Labels: ['W1','W2'] } }
            if(data.HRChart || data.hrChart) {
                const hr = data.HRChart || data.hrChart;
                hrChart.updateSeries([
                    { name: 'Attrition', data: hr.Attrition },
                    { name: 'Long Leave', data: hr.Leaves },
                    { name: 'TLOA', data: hr.TLOA }
                ]);
                hrChart.updateOptions({ xaxis: { categories: hr.Labels }});
            }

            // 3. Update Table
            const tableRows = data.TableData || data.tableData;
            mainTable.clear().rows.add(tableRows).draw();
        }

        // --- INITIALIZATION ---
        function initCharts() {
            // 1. Trend
            trendChart = new ApexCharts(document.querySelector("#chart-trend"), {
                ...baseChartConfig,
                series: [], // Starts empty
                chart: { type: 'area', height: 320, toolbar:{show:false}, background: 'transparent' },
                colors: ['#3699FF', '#0BB783'],
                fill: { type: 'gradient', gradient: { opacityFrom: 0.6, opacityTo: 0.1 } },
                stroke: { width: 2, curve: 'smooth' }
            });
            trendChart.render();

            // 2. Geo
            geoChart = new ApexCharts(document.querySelector("#chart-geo"), {
                ...baseChartConfig,
                series: [],
                labels: [],
                chart: { type: 'donut', height: 280, background: 'transparent' },
                colors: ['#3699FF', '#8950FC', '#0BB783', '#FFA800'],
                stroke: { width: 0 },
                plotOptions: { pie: { donut: { labels: { show: true, total: { show: true, color: '#fff' } } } } },
                legend: { position: 'bottom' }
            });
            geoChart.render();

            // 3. Delta Bar
            deltaChart = new ApexCharts(document.querySelector("#chart-delta"), {
                ...baseChartConfig,
                series: [],
                chart: { type: 'bar', height: 250, background: 'transparent' },
                plotOptions: { bar: { horizontal: true, borderRadius: 4, barHeight: '50%' } },
                colors: ['#F64E60'],
                xaxis: { categories: [], labels: { style: { colors: '#92929f' } } }
            });
            deltaChart.render();

            // 4. Billable Radial
            billChart = new ApexCharts(document.querySelector("#chart-billable"), {
                ...baseChartConfig,
                series: [0],
                chart: { type: 'radialBar', height: 250, background: 'transparent' },
                plotOptions: {
                    radialBar: {
                        hollow: { size: '70%' },
                        track: { background: '#2B2B40' },
                        dataLabels: {
                            name: { show: false },
                            value: { color: '#fff', fontSize: '24px', show: true, formatter: val => val + "%" }
                        }
                    }
                },
                colors: ['#0BB783'],
                stroke: { lineCap: 'round' }
            });
            billChart.render();

            // 5. HR Stacked
            hrChart = new ApexCharts(document.querySelector("#chart-hr"), {
                ...baseChartConfig,
                series: [],
                chart: { type: 'bar', height: 250, stacked: true, background: 'transparent' },
                colors: ['#F64E60', '#FFA800', '#8950FC'],
                plotOptions: { bar: { columnWidth: '40%' } }
            });
            hrChart.render();
        }

        function initTable() {
            mainTable = $('#mainTable').DataTable({
                data: [], // Starts empty
                pageLength: 5,
                dom: 'rtip',
                columns: [
                    {
                        // Fixed Double Icon Issue: Using custom-drill
                        className: 'custom-drill text-center',
                        data: null, orderable: false,
                        defaultContent: '<i class="fas fa-plus-circle text-primary" style="cursor:pointer; font-size: 1.1rem;"></i>'
                    },
                    { data: 'Week' },
                    { data: 'Geo' },
                    { data: 'Client' },
                    { data: 'Project' },
                    { data: 'SchHC', className: 'text-end' },
                    { data: 'ActSchHC', className: 'text-end' },
                    {
                        data: 'Delta', className: 'text-center',
                        render: function (d) {
                            return d > 0 
                            ? `<span class="badge bg-danger bg-opacity-25 text-danger border border-danger border-opacity-25">${d}</span>` 
                            : `<span class="text-muted">-</span>`;
                        }
                    },
                    {
                        data: 'Delta', className: 'text-center',
                        render: function (d) {
                            return d === 0 
                            ? '<span class="badge bg-success bg-opacity-10 text-success">On Track</span>' 
                            : '<span class="badge bg-warning bg-opacity-10 text-warning">Review</span>';
                        }
                    }
                ]
            });

            // Drilldown Interaction
            $('#mainTable tbody').on('click', 'td.custom-drill', function () {
                var tr = $(this).closest('tr');
                var row = mainTable.row(tr);
                var icon = $(this).find('i');

                if (row.child.isShown()) {
                    row.child.hide();
                    tr.removeClass('shown');
                    icon.removeClass('fa-minus-circle').addClass('fa-plus-circle');
                } else {
                    var d = row.data();
                    // Sober Design for Drilldown
                    var content = `
                    <div class="drilldown-container">
                        <div class="row">
                            <div class="col-md-3"><span class="drill-label">Program</span><div class="drill-val">${d.Program}</div></div>
                            <div class="col-md-3"><span class="drill-label">OU Desc</span><div class="drill-val">${d.OUDescription}</div></div>
                            <div class="col-md-3"><span class="drill-label">Billable</span>
                                <div class="drill-val">${d.BillableStatus === 'Yes' ? '<span class="text-success">Billable</span>' : '<span class="text-secondary">Non-Billable</span>'}</div>
                            </div>
                            <div class="col-md-3 text-end"><button class="btn btn-sm btn-outline-primary">Details</button></div>
                        </div>
                    </div>`;
                    row.child(content).show();
                    tr.addClass('shown');
                    icon.removeClass('fa-plus-circle').addClass('fa-minus-circle');
                }
            });
        }
    </script>
